# 快速开始：阶段一（D-2 返院前接收与基线重建）

## 1. 阶段目标

将医院专业术语转换为机构可执行指令，完成住院前后状态差异识别，为返院承接做准备。

调用链（业务视角）：

`A0 → A4 → A5 → A6 → A2/A3`

工作流说明：

1. `A0` 识别新返院病例，创建病例实例（当前作为外部接入步骤，手工或外部系统完成）。
2. `A4` 解析出院摘要/医嘱/干预摘要，生成《出院摘要结构化包》。
3. `A5` 拉取住院前 ECR 与出院状态对比，生成《ECR状态对比报告》。
4. `A6` 基于结构化包、对比报告、护理规则，生成《返院适应期专项任务包》。
5. `A2` 接收护士核查项；`A3` 接收 GP 关注项与高风险摘要。
6. 人工确认：护士确认承接方案，GP 确认高风险干预。

---

## 2. 关键输入与输出

关键输入：

- 客户基本画像
- 出院摘要原文 / 图片 / OCR
- 出院医嘱
- 住院前 ECR 摘要

关键输出：

- 《出院摘要结构化包》
- 《ECR状态对比报告》
- 《BPSD返院适应期专项任务包》

> 说明：当前工程中 A6 的默认文件名为《返院适应期专项任务包》。如果你希望输出文件名严格改成《BPSD返院适应期专项任务包》，可在 `agent_a6_task_pack_agent.py` 中调整 `pdf_title` 与 `output_file_stem`。

## 3. 最小可运行代码（LangGraph）

将下面脚本保存为 `/srv/supercare/task-agent/quickstart_d2_workflow.py` 后执行。

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-

from typing import Any, Dict, TypedDict

from langgraph.graph import END, START, StateGraph

from agent_a2_cca_hub import get_langgraph_node as node_a2
from agent_a3_cda_hub import get_langgraph_node as node_a3
from agent_a4_discharge_parser import get_langgraph_node as node_a4
from agent_a5_baseline_compare import get_langgraph_node as node_a5
from agent_a6_task_pack_agent import get_langgraph_node as node_a6
from common_utils import DEFAULT_CONFIG_PATH, DEFAULT_GRAPH_PATH


class D2State(TypedDict, total=False):
    # 基础运行参数
    graph_path: str
    config_path: str
    use_tools: bool

    # 各节点输入
    a4_input: str
    a5_input: str
    a6_input: str
    a2_input: str
    a3_input: str

    # 各节点输出
    a4_output: Dict[str, Any]
    a5_output: Dict[str, Any]
    a6_output: Dict[str, Any]
    a2_output: Dict[str, Any]
    a3_output: Dict[str, Any]


def merge_previous_outputs(state: D2State, current_business_input: str) -> str:
    """把前序节点输出汇总后，拼接到当前节点输入。"""
    history_sections = []
    for agent_key in ("a4_output", "a5_output", "a6_output", "a2_output", "a3_output"):
        output = state.get(agent_key, {})
        if not output:
            continue
        history_sections.append(
            f"[{agent_key}]\n"
            f"success={output.get('success')}\n"
            f"text={output.get('text', '')}\n"
            f"pdf_path={output.get('pdf_path', '')}"
        )
    history_text = "\n\n".join(history_sections) if history_sections else "（暂无前序输出）"
    return f"【当前任务输入】\n{current_business_input}\n\n【前序节点输出汇总】\n{history_text}"


def prepare_a5_input(state: D2State) -> D2State:
    return {**state, "a5_input": merge_previous_outputs(state, state.get("a5_input", ""))}


def prepare_a6_input(state: D2State) -> D2State:
    return {**state, "a6_input": merge_previous_outputs(state, state.get("a6_input", ""))}


def prepare_a2_input(state: D2State) -> D2State:
    return {**state, "a2_input": merge_previous_outputs(state, state.get("a2_input", ""))}


def prepare_a3_input(state: D2State) -> D2State:
    return {**state, "a3_input": merge_previous_outputs(state, state.get("a3_input", ""))}


def create_d2_workflow():
    workflow = StateGraph(D2State)

    # 注册节点
    workflow.add_node("a4", node_a4())
    workflow.add_node("prepare_a5_input", prepare_a5_input)
    workflow.add_node("a5", node_a5())
    workflow.add_node("prepare_a6_input", prepare_a6_input)
    workflow.add_node("a6", node_a6())
    workflow.add_node("prepare_a2_input", prepare_a2_input)
    workflow.add_node("a2", node_a2())
    workflow.add_node("prepare_a3_input", prepare_a3_input)
    workflow.add_node("a3", node_a3())

    # 定义链路：每个节点执行前，先拼接前序输出
    workflow.add_edge(START, "a4")
    workflow.add_edge("a4", "prepare_a5_input")
    workflow.add_edge("prepare_a5_input", "a5")
    workflow.add_edge("a5", "prepare_a6_input")
    workflow.add_edge("prepare_a6_input", "a6")
    workflow.add_edge("a6", "prepare_a2_input")
    workflow.add_edge("a6", "prepare_a3_input")
    workflow.add_edge("prepare_a2_input", "a2")
    workflow.add_edge("prepare_a3_input", "a3")
    workflow.add_edge("a2", END)
    workflow.add_edge("a3", END)

    return workflow.compile()


if __name__ == "__main__":
    app = create_d2_workflow()

    # A0 步骤（外部系统/人工）完成后，先提供每个节点的当前任务输入
    # 工作流会自动追加“前序节点输出汇总”
    initial_state: D2State = {
        "graph_path": str(DEFAULT_GRAPH_PATH),
        "config_path": str(DEFAULT_CONFIG_PATH),
        "use_tools": True,
        "a4_input": (
            "患者陈女士，女，上海籍，小学学历，丧偶，育有1名子女。"
            "因头部外伤急诊入院，CT提示右额部皮下血肿。"
            "出院医嘱：规律服用降压、调脂药物，每日监测血压，做好跌倒防护。"
        ),
        "a5_input": (
            "住院前ECR摘要：阿尔茨海默病性痴呆，血压偏高、血脂异常，可独立进食穿衣，"
            "平地行走需部分辅助，存在拒绝配合康复的BPSD症状。"
        ),
        "a6_input": (
            "请基于结构化包与对比报告，输出返院适应期专项任务包，"
            "需覆盖用药指导、跌倒防护、认知干预、家属支持，并明确频次。"
        ),
        "a2_input": "请给出护士核查项：D5录入规范、风险观察点、周重点任务。",
        "a3_input": "请给出GP关注项与高风险摘要：是否需精神科会诊、会诊触发条件、目标更新建议。",
    }

    result_state = app.invoke(initial_state)

    # 打印关键产物路径（PDF）
    print("A4 PDF:", result_state.get("a4_output", {}).get("pdf_path"))
    print("A5 PDF:", result_state.get("a5_output", {}).get("pdf_path"))
    print("A6 PDF:", result_state.get("a6_output", {}).get("pdf_path"))
    print("A2 PDF:", result_state.get("a2_output", {}).get("pdf_path"))
    print("A3 PDF:", result_state.get("a3_output", {}).get("pdf_path"))
```

## 4. 运行方式

在 `/srv/supercare/task-agent` 目录执行：

```bash
python3 quickstart_d2_workflow.py
```

成功后可在 `/srv/supercare/task-agent/output` 查看阶段一产物：

- `a4_出院摘要结构化包.pdf`
- `a5_ECR状态对比报告.pdf`
- `a6_返院适应期专项任务包.pdf`
- （支撑协作输出）`a2_护士协作专业答复.pdf`
- （支撑协作输出）`a3_GP协作专业答复.pdf`

## 5. 实施建议（上线前）

- 若出院摘要来自图片/OCR，先做人工质检，避免关键信息识别错误。
- 对“无依据留空”字段建立人工补录清单（尤其评分、复诊时间、触发因素）。
- 护士与 GP 确认后，将最终承接方案回写到机构主系统，形成闭环。
